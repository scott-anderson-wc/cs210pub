Exercise 3

Here is one solution: 

var num = parseFloat(prompt("Enter any num"));

if (num > 0) {
  alert("the number is positive");
} else if (num < 0) {
  alert("the number is negative"); 
} else {
  alert("the number is zero");
}

Here's another: 

var num = parseFloat(prompt("Enter any num"));

if (num > 0) {
  alert("the number is positive");
} else if (num == 0) {
  alert("the number is zero"); 
} else {
  alert("the number is negative");
}

There are many other solutions that can be obtained by permuting the order of the tests.

Exercise 6

  1. Here's a fleshed-out version of the Rock/paper/scissors code: 
    var p1 = prompt("Player 1: enter one of rock,paper,scissors","rock");
var p2 = prompt("Player 2: enter one of rock,paper,scissors","rock");

if (p1 == p2) {
  alert("players tie");
} else if ( ((p1 == "rock") && (p2 == "scissors"))
            || ((p1 == "scissors") && (p2 == "paper"))
            || ((p1 == "paper") && (p2 == "rock")) 
            ) {
  alert("player 1 wins");
} else {
  alert ("player 2 wins");
}
  2. Here's a version that performs a sanity check on the inputs and won't play the game if they're bad. 
    var p1 = prompt("Player 1: enter one of rock, paper, scissors", "rock");
if (! ((p1 == "rock") || (p1 == "paper") || (p1 == "scissors")) ) { // if #1
  alert(p1 + " isn't a valid choice. Aborting the game!");
} else { // p1 is valid; continue the game
  var p2 = prompt("Player 2: enter one of rock,paper,scissors","rock");
  if (! ((p2 == "rock") || (p2 == "paper") || (p2 == "scissors")) ) { // if #2
    alert(p2 + " isn't a valid choice. Aborting the game!");
  } else { // p2 is valid; continue the game
    // This is the same code as in Part 1:
    if (p1 == p2) { // if #3
      alert("players tie");
    } else if ( ((p1 == "rock") && (p2 == "scissors"))
                || ((p1 == "scissors") && (p2 == "paper"))
                || ((p1 == "paper") && (p2 == "rock")) 
                ) {
      alert("player 1 wins");
    } else {
      alert ("player 2 wins");
    } // end if #3
  } // end if #2
} // end if #1
